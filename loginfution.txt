// Dynamic import for face-api to avoid SSR issues
const faceapi = typeof window !== "undefined" ? require('@vladmandic/face-api/dist/face-api.esm.js') : null;

const SECRET_KEY = "NEOTALK_AES_256_KEY";

interface Message {
  id: string;
  senderName: string;
  text: string;
  timestamp: Date;
  isUser: boolean;
  isSystem?: boolean;
  isStreaming?: boolean;
}

export default function App() {
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [username, setUsername] = useState("");
  const [socket, setSocket] = useState<Socket | null>(null);

  useEffect(() => {
    // Dynamically connect to the same host on port 5000
    const socketHost = window.location.hostname;
    const newSocket = io(`http://${socketHost}:5000`);
    setSocket(newSocket);
    return () => { newSocket.close(); };
  }, []);

  const handleAuth = (name: string) => {
    setUsername(name);
    setIsAuthenticated(true);
  };

  return (
    <div className="bg-[#0b0f19] text-white font-sans overflow-hidden h-screen w-full selection:bg-emerald-500/30">
      <AnimatePresence mode="wait">
        {!isAuthenticated ? (
          <BiometricLogin key="login" socket={socket} onAuth={handleAuth} />
        ) : (
          <MainDashboard key="dashboard" socket={socket} username={username} />
        )}
      </AnimatePresence>
    </div>
  );
}

function BiometricLogin({ socket, onAuth }: { socket: Socket | null, onAuth: (name: string) => void }) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [scanStatus, setScanStatus] = useState("LOADING BIOMETRIC MODELS...");
  const [isScanning, setIsScanning] = useState(false);
  const [scanStage, setScanStage] = useState<"loading" | "idle" | "align" | "left" | "right" | "success" | "error">("loading");
  const [progress, setProgress] = useState(0);
  const [modelsLoaded, setModelsLoaded] = useState(false);

  const speak = (text: string) => {
    window.speechSynthesis.cancel();
    const msg = new SpeechSynthesisUtterance(text);
    msg.rate = 0.9;
    msg.pitch = 1.1;
    window.speechSynthesis.speak(msg);
  };

  useEffect(() => {
    const loadModels = async () => {
      if (modelsLoaded) return;
      const MODEL_URL = 'https://cdn.jsdelivr.net/npm/@vladmandic/face-api/model/';
      await faceapi.nets.tinyFaceDetector.loadFromUri(MODEL_URL);
      await faceapi.nets.faceLandmark68Net.loadFromUri(MODEL_URL);
      setModelsLoaded(true);
      setScanStatus("TAP TO INITIATE SECURE LINK");
      setScanStage("idle");
    };
    loadModels();
  }, [modelsLoaded]);

  useEffect(() => {
    if (!socket) return;
    const onSuccess = (data: any) => {
      setTimeout(() => onAuth(data.username), 1500);
    };
    socket.on("auth_success", onSuccess);
    return () => { socket.off("auth_success", onSuccess); };
  }, [socket, onAuth]);

  const startScan = async () => {
    if (!socket || !modelsLoaded || isScanning) return;
    
    try {
      setScanStatus("NODE SEARCH INITIATED...");
      
      // NUCLEAR RESET: Force-kill any lingering tracks and clear the video source
      if (videoRef.current && videoRef.current.srcObject) {
        const tracks = (videoRef.current.srcObject as MediaStream).getTracks();
        tracks.forEach(t => {
            t.stop();
            t.enabled = false;
        });
        videoRef.current.srcObject = null;
      }

      if (!navigator.mediaDevices) throw new Error("HARDWARE_API_UNAVAILABLE");

      let stream: MediaStream | null = null;
      
      // LAYER 1: Basic acquisition (Most compatible)
      try {
        stream = await navigator.mediaDevices.getUserMedia({ video: true });
      } catch (e1) {
        console.warn("Layer 1 fail, trying Layer 2...");
        // LAYER 2: Facing mode acquisition
        try {
          stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } });
        } catch (e2) {
          console.warn("Layer 2 fail, trying Device Enumeration...");
          // LAYER 3: Manual device loop (Hunts for a free camera chip)
          const devices = await navigator.mediaDevices.enumerateDevices();
          const videoInputs = devices.filter(d => d.kind === 'videoinput');
          
          for (const dev of videoInputs) {
            try {
              stream = await navigator.mediaDevices.getUserMedia({ 
                video: { deviceId: { exact: dev.deviceId } } 
              });
              if (stream) break;
            } catch (e3) { continue; }
          }
        }
      }

      if (!stream) throw new Error("NO_AVAILABLE_CAMERA");
      
      setIsScanning(true);
      if (videoRef.current) {
        videoRef.current.srcObject = stream;
        // Small delay to let hardware sync with the OS
        await new Promise(r => setTimeout(r, 200));
        await videoRef.current.play();
      }
      
      setScanStage("align");
      setScanStatus("LINK SECURED. ALIGNING...");
      speak("Secure link established.");
    } catch (err: any) {
      console.error("Hardware Node Error:", err);
      setIsScanning(false);
      
      if (!navigator.mediaDevices) {
        setScanStatus("SECURE CONTEXT REQUIRED (SSL)");
        speak("Biometric hardware requires an encrypted HTTPS connection. Please use localhost or SSL.");
      } else if (err.name === "NotAllowedError") {
        setScanStatus("ACCESS BLOCKED BY BROWSER");
        speak("Camera access is blocked. Please click the lock icon in your browser and reset permissions.");
      } else if (err.message === "NO_AVAILABLE_CAMERA") {
        setScanStatus("ALL CAMERA NODES BUSY");
        speak("All camera nodes are busy. Please close other apps using the camera.");
      } else {
        setScanStatus("HARDWARE LINK FAILED");
        speak("Hardware nodes are not responding. You can use Emergency Access below.");
      }
      return;
    }

    let currentStage = "align";
    let p = 0;
    let lastRatio = 1.0;
    let lastNoseX = 0;
    let lastNoseY = 0;
    let faceMissingCount = 0;

    const detectLoop = async () => {
      if (!videoRef.current) return;
      const detection = await faceapi.detectSingleFace(videoRef.current, new faceapi.TinyFaceDetectorOptions()).withFaceLandmarks();

      if (!detection || detection.detection.score < 0.1) {
        faceMissingCount++;
        if (faceMissingCount > 450) { // Very tolerant (7.5 seconds)
          if (currentStage !== "error") {
            setScanStage("error");
            setScanStatus("LINK WEAK. HOLD STILL.");
            if (faceMissingCount > 900) { p = 0; setProgress(0); currentStage = "error"; }
          }
        }
      } else {
        faceMissingCount = 0;
        const nose = detection.landmarks.getNose()[0];
        const leftEye = detection.landmarks.getLeftEye()[0];
        const rightEye = detection.landmarks.getRightEye()[0];
        const leftDist = Math.abs(nose.x - leftEye.x);
        const rightDist = Math.abs(nose.x - rightEye.x);
        const ratio = leftDist / rightDist;

        if (currentStage === "error") {
          currentStage = "align";
          setScanStage("align");
          setScanStatus("LINK RE-ESTABLISHED");
        }

        if (currentStage === "align") {
          // STEP 1: CENTER & LIVENESS (0 - 25%)
          const movement = Math.abs(ratio - lastRatio) + (Math.abs(nose.x - lastNoseX) / 30);
          if (movement > 0.008) { p = Math.min(25, p + 2.5); } 
          else { p = Math.min(25, p + 0.3); }
          setProgress(p);
          if (p >= 25) {
            currentStage = "left";
            setScanStage("left");
            setScanStatus("STEP 2: TURN HEAD LEFT");
            speak("Turn your head left.");
            lastRatio = ratio;
          }
        } else if (currentStage === "left") {
          // STEP 2: TURN LEFT (25 - 50%)
          if (ratio < 0.88 || ratio > 1.15) { 
            p = Math.min(50, p + 2.5); 
            setProgress(p); 
          }
          if (p >= 50) {
            currentStage = "right";
            setScanStage("right");
            setScanStatus("STEP 3: TURN HEAD RIGHT");
            speak("Turn your head right.");
            lastRatio = ratio;
          }
        } else if (currentStage === "right") {
          // STEP 3: TURN RIGHT (50 - 75%)
          if (ratio > 1.2 || ratio < 0.8) { 
            p = Math.min(75, p + 2.5); 
            setProgress(p); 
          }
          if (p >= 75) {
            currentStage = "rotate";
            setScanStage("rotate");
            setScanStatus("STEP 4: ROTATE HEAD");
            speak("Slowly rotate your head.");
            lastRatio = ratio;
          }
        } else if (currentStage === "rotate") {
          // STEP 4: ROTATE (75 - 100%)
          const movement = Math.abs(ratio - lastRatio) + (Math.abs(nose.x - lastNoseX) / 25);
          if (movement > 0.01 || ratio > 1.3 || ratio < 0.7) { 
            p = Math.min(100, p + 3.0); 
            setProgress(p); 
          }
          if (p >= 100) {
            currentStage = "success";
            setScanStage("success");
            setScanStatus("IDENTITY SECURED. LOGGING IN.");
            speak("Identity secured. Welcome to Aura.");
            socket.emit("biometric_auth", {});
            return;
          }
        }

        lastRatio = ratio;
        lastNoseX = nose.x;
        lastNoseY = nose.y;
      }
      requestAnimationFrame(detectLoop);
    };
    detectLoop();
  };

  const radius = 140;
  const circumference = 2 * Math.PI * radius;
  const strokeDashoffset = circumference - (progress / 100) * circumference;

  return (
    <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ scale: 1.1, opacity: 0 }} className="flex flex-col items-center justify-center h-full bg-[#050810]">
      <div className="mb-12 text-center">
        <h1 className="text-3xl font-black tracking-widest text-transparent bg-clip-text bg-gradient-to-r from-emerald-400 to-cyan-500 mb-2">NEOTALK SECURE</h1>
        <p className="text-gray-500 font-mono text-sm tracking-widest uppercase">Biometric Authentication</p>
      </div>

      <motion.div
        animate={scanStage === "error" ? { x: [-15, 15, -15, 15, -10, 10, 0] } : {}}
        transition={{ duration: 0.5 }}
        className="relative group cursor-pointer flex items-center justify-center w-80 h-80" onClick={startScan}
      >

        {/* SVG Ring Overlay */}
        <svg className="absolute inset-0 w-full h-full transform -rotate-90 pointer-events-none z-10" viewBox="0 0 320 320">
          <defs>
            <mask id="progress-mask">
              <rect x="0" y="0" width="320" height="320" fill="black" />
              <circle
                cx="160" cy="160" r={radius}
                fill="none" stroke="white" strokeWidth="16"
                strokeDasharray={`${circumference} ${circumference}`}
                strokeDashoffset={strokeDashoffset}
                className="transition-all duration-100 ease-linear"
              />
            </mask>
          </defs>

          {/* Background Dotted Ring */}
          <circle cx="160" cy="160" r={radius} className="fill-none stroke-[8px] transition-colors duration-500 stroke-white/10" strokeDasharray="6 8" />

          {/* Foreground Animated Ring (Dotted & Masked) */}
          {(scanStage === 'align' || scanStage === 'left' || scanStage === 'right' || scanStage === 'rotate' || scanStage === 'success' || scanStage === 'error') && (
            <circle
              cx="160" cy="160" r={radius}
              className={`fill-none stroke-[8px] transition-colors duration-300 ${scanStage === 'error' ? 'stroke-red-500' : 'stroke-emerald-500'}`}
              strokeDasharray="6 8"
              mask="url(#progress-mask)"
            />
          )}
        </svg>

        {/* Vertical Scanner Line during 'align' phase */}
        {scanStage === 'align' && (
          <div className="absolute inset-0 w-full h-[3px] bg-emerald-400 shadow-[0_0_20px_rgba(52,211,153,1)] z-20 pointer-events-none rounded-full"
            style={{ animation: 'scanline 2s ease-in-out infinite' }}>
          </div>
        )}

        <style>{`
            @keyframes scanline {
                0% { top: 0%; opacity: 0; }
                10% { opacity: 1; }
                50% { top: 100%; }
                90% { opacity: 1; }
                100% { top: 0%; opacity: 0; }
            }
        `}</style>

        {/* Circular Video Mask */}
        <div className={`relative w-[270px] h-[270px] rounded-full overflow-hidden shadow-2xl transition-all duration-500 ${scanStage === 'success' ? 'shadow-[0_0_50px_rgba(16,185,129,0.5)]' : 'shadow-cyan-500/10'}`}>
          <video ref={videoRef} autoPlay playsInline muted className={`w-full h-full object-cover transition-all duration-700 ${isScanning ? "scale-110 brightness-125 contrast-110" : "grayscale opacity-50"}`}></video>

          {!isScanning && (
            <div className="absolute inset-0 flex flex-col items-center justify-center bg-black/60 backdrop-blur-md hover:bg-black/40 transition-all">
              {scanStage === "loading" ? (
                <div className="w-12 h-12 border-4 border-cyan-500 border-t-transparent rounded-full animate-spin mb-2"></div>
              ) : (
                <ScanFace className="w-16 h-16 text-cyan-400 mb-2 animate-pulse" />
              )}
            </div>
          )}
        </div>
      </motion.div>

      <div className="mt-12 text-center h-16">
        <motion.div
          key={scanStatus}
          initial={{ opacity: 0, y: 10 }}
          animate={{ opacity: 1, y: 0 }}
          className={`font-mono tracking-widest text-sm font-bold ${scanStage === 'success' ? 'text-emerald-400' : 'text-cyan-400'}`}

        >
          {scanStatus}
        </motion.div>
        {isScanning && scanStage !== 'success' && scanStage !== 'error' && (
          <p className="text-gray-500 text-xs font-mono mt-2 uppercase tracking-widest">Keep your device level</p>
        )}

        {/* Prominent Emergency Bypass */}
        <button
          onClick={() => {
            speak("Emergency bypass activated. Welcome Agent.");
            socket?.emit("biometric_auth", {});
          }}
          className="mt-12 px-6 py-2 border border-emerald-500/20 rounded-full text-[10px] text-emerald-500/40 hover:text-emerald-400 hover:bg-emerald-500/5 transition-all font-mono uppercase tracking-[0.2em] animate-pulse"
        >
          Emergency Dashboard Access
        </button>
      </div>
    </motion.div>
  );
}

function MainDashboard({ socket, username }: { socket: Socket | null, username: string }) {
  const [activeTab, setActiveTab] = useState("chat");
  const [onlineUsers, setOnlineUsers] = useState<any[]>([]);
  const [incomingCall, setIncomingCall] = useState<{ signal: any, from: string, callerName: string } | null>(null);
  const [activeCall, setActiveCall] = useState<{ userId: string, username: string, isCaller: boolean, initialSignal?: any } | null>(null);

  useEffect(() => {
    if (!socket) return;
    socket.on("update_users", (users) => setOnlineUsers(users));

    socket.on("incoming_call", (data) => {
      setIncomingCall(data);
    });

    socket.on("call_ended", () => {
      setActiveCall(null);
      setIncomingCall(null);
    });

    return () => {
      socket.off("update_users");
      socket.off("incoming_call");
      socket.off("call_ended");
    }
  }, [socket]);

  const startCall = (user: any) => {
    setActiveCall({ userId: user.id, username: user.username, isCaller: true });
  };

  const acceptCall = () => {
    if (!incomingCall) return;
    setActiveCall({ userId: incomingCall.from, username: incomingCall.callerName, isCaller: false, initialSignal: incomingCall.signal });
    setIncomingCall(null);
  };

  const declineCall = () => {
    if (incomingCall) {
      socket?.emit("end_call", { to: incomingCall.from });
      setIncomingCall(null);
    }
  };

  return (
    <div className="flex h-screen bg-[#090d16]">
      {/* Sidebar Dashboard Navigation */}
      <nav className="w-20 md:w-64 border-r border-white/5 bg-[#050810] flex flex-col">
        <div className="h-20 flex items-center justify-center md:justify-start md:px-6 border-b border-white/5">
          <Cpu className="w-8 h-8 text-emerald-500" />
          <span className="hidden md:block ml-3 font-black tracking-widest text-transparent bg-clip-text bg-gradient-to-r from-emerald-400 to-cyan-500">NEOTALK</span>
        </div>

        <div className="flex-1 py-6 flex flex-col gap-2 px-3">
          <NavItem icon={<MessageSquare />} label="Global Chat" active={activeTab === 'chat'} onClick={() => setActiveTab('chat')} />
          <NavItem icon={<Users />} label="Online Users" active={activeTab === 'users'} onClick={() => setActiveTab('users')} />
          <NavItem icon={<LayoutDashboard />} label="Admin Stats" active={activeTab === 'admin'} onClick={() => setActiveTab('admin')} />
        </div>

        <div className="p-4 border-t border-white/5 flex items-center justify-center md:justify-start gap-3">
          <div className="w-10 h-10 rounded-full bg-gradient-to-br from-emerald-500 to-cyan-600 p-[2px]">
            <div className="w-full h-full bg-[#050810] rounded-full flex items-center justify-center">
              <img src={`https://api.dicebear.com/7.x/bottts/svg?seed=${username}`} alt="avatar" className="w-8 h-8 rounded-full" />
            </div>
          </div>
          <div className="hidden md:block">
            <p className="text-sm font-bold text-white">{username}</p>
            <p className="text-[10px] text-emerald-500 font-mono">Encrypted Session</p>
          </div>
        </div>
      </nav>

      {/* Main Content Area */}
      <main className="flex-1 flex flex-col relative">
        {activeTab === 'chat' && <MultiplayerChat socket={socket} username={username} />}
        {activeTab === 'admin' && <AdminPlaceholder />}
        {activeTab === 'users' && <OnlineUsersDirectory onlineUsers={onlineUsers} onCall={startCall} />}

        {/* Incoming Call Overlay */}
        <AnimatePresence>
          {incomingCall && !activeCall && (
            <motion.div initial={{ y: -100, opacity: 0 }} animate={{ y: 0, opacity: 1 }} exit={{ y: -100, opacity: 0 }} className="absolute top-10 left-1/2 -translate-x-1/2 bg-[#0c1222] border border-emerald-500 shadow-[0_0_50px_rgba(16,185,129,0.3)] rounded-2xl p-6 z-50 flex items-center gap-6">
              <div className="w-16 h-16 rounded-full bg-emerald-500/20 flex items-center justify-center animate-pulse">
                <img src={`https://api.dicebear.com/7.x/bottts/svg?seed=${incomingCall.callerName}`} className="w-12 h-12 rounded-full" />
              </div>
              <div>
                <h3 className="text-xl font-bold text-white">Incoming Secure Call</h3>
                <p className="text-emerald-500 font-mono text-sm">{incomingCall.callerName} is requesting a video link...</p>
              </div>
              <div className="flex gap-3 ml-4">
                <button onClick={declineCall} className="w-12 h-12 rounded-full bg-red-500/20 text-red-500 hover:bg-red-500 hover:text-white transition-all flex items-center justify-center"><X className="w-6 h-6" /></button>
                <button onClick={acceptCall} className="w-12 h-12 rounded-full bg-emerald-500 text-[#050810] hover:bg-emerald-400 transition-all flex items-center justify-center shadow-[0_0_20px_rgba(16,185,129,0.5)]"><PhoneCall className="w-6 h-6" /></button>
              </div>
            </motion.div>
          )}
        </AnimatePresence>

        {/* Active Call Interface */}
        <AnimatePresence>
          {activeCall && (
            <VideoCallInterface
              socket={socket}
              activeCall={activeCall}
              myUsername={username}
              onEnd={() => setActiveCall(null)}
            />
          )}
        </AnimatePresence>
      </main>
    </div>
  );
}

function MultiplayerChat({ socket, username }: { socket: Socket | null, username: string }) {
  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState("");
  const messagesEndRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!socket) return;

    socket.on("receive_message", (msg) => {
      let decryptedText = msg.text;
      if (msg.isEncrypted) {
        try {
          const bytes = CryptoJS.AES.decrypt(msg.text, SECRET_KEY);
          decryptedText = bytes.toString(CryptoJS.enc.Utf8);
        } catch (err) {
          decryptedText = "[ENCRYPTED CONTENT UNREADABLE]";
        }
      }
      setMessages(p => [...p, { ...msg, text: decryptedText }]);
    });

    socket.on("system_broadcast", (msg) => setMessages(p => [...p, { id: Math.random().toString(), senderName: "SYSTEM", text: msg.text, timestamp: new Date(), isUser: false, isSystem: true }]));

    socket.on("stream_start", (data) => {
      setMessages(p => [...p, { id: data.messageId, senderName: data.senderName, text: "", timestamp: new Date(), isUser: data.isUser, isStreaming: true }]);
    });
    socket.on("stream_chunk", (data) => {
      setMessages(p => p.map(m => m.id === data.messageId ? { ...m, text: m.text + data.text } : m));
    });
    socket.on("stream_end", (data) => {
      setMessages(p => p.map(m => m.id === data.messageId ? { ...m, isStreaming: false } : m));
    });

    return () => {
      socket.off("receive_message"); socket.off("system_broadcast");
      socket.off("stream_start"); socket.off("stream_chunk"); socket.off("stream_end");
    };
  }, [socket]);

  useEffect(() => messagesEndRef.current?.scrollIntoView({ behavior: "smooth" }), [messages]);

  const handleSend = () => {
    if (!input.trim() || !socket) return;

    // AES Encryption using CryptoJS before sending over the network
    const encryptedText = CryptoJS.AES.encrypt(input, SECRET_KEY).toString();

    // We send both the encrypted text for users, and the raw text for the AI backend hook
    socket.emit("send_message", { text: encryptedText, decryptedTextForAi: input });
    setInput("");
  };

  return (
    <div className="flex flex-col h-full w-full">
      {/* WhatsApp/Discord Style Header */}
      <header className="h-20 flex items-center justify-between px-6 border-b border-white/5 bg-[#090d16]">
        <div className="flex items-center gap-4">
          <div className="w-12 h-12 rounded-full bg-emerald-500/20 flex items-center justify-center border border-emerald-500/30">
            <Lock className="w-6 h-6 text-emerald-400" />
          </div>
          <div>
            <h1 className="text-lg font-bold text-white">Global Encrypted Channel</h1>
            <p className="text-xs text-emerald-500 font-mono flex items-center gap-1"><ShieldCheck className="w-3 h-3" /> End-to-End Encrypted (AES-256)</p>
          </div>
        </div>
        <div className="flex gap-4">
          <button className="w-10 h-10 rounded-full hover:bg-white/5 flex items-center justify-center text-gray-400 transition-all"><Video className="w-5 h-5" /></button>
          <button className="w-10 h-10 rounded-full hover:bg-white/5 flex items-center justify-center text-gray-400 transition-all"><Phone className="w-5 h-5" /></button>
          <button className="w-10 h-10 rounded-full hover:bg-white/5 flex items-center justify-center text-gray-400 transition-all"><MoreVertical className="w-5 h-5" /></button>
        </div>
      </header>

      {/* Chat Area */}
      <div className="flex-1 overflow-y-auto p-6 space-y-6 bg-[url('https://www.transparenttextures.com/patterns/cubes.png')] bg-[#050810] bg-opacity-95 bg-blend-multiply scrollbar-thin scrollbar-thumb-emerald-900/50">
        <AnimatePresence>
          {messages.length === 0 && (
            <div className="text-center mt-20 text-gray-600 font-mono text-sm uppercase tracking-widest bg-white/5 inline-block px-6 py-2 rounded-full mx-auto block w-max">
              <ShieldCheck className="inline-block w-4 h-4 mr-2 text-emerald-500" /> Messages are end-to-end encrypted
            </div>
          )}
          {messages.map((msg) => (
            <motion.div key={msg.id} initial={{ opacity: 0, scale: 0.95 }} animate={{ opacity: 1, scale: 1 }} className={`flex w-full ${msg.senderName === username ? "justify-end" : msg.isSystem ? "justify-center" : "justify-start"}`}>
              {msg.isSystem ? (
                <div className="text-[10px] font-mono text-emerald-600 tracking-widest bg-emerald-950/20 px-4 py-2 rounded-full border border-emerald-900/50 uppercase my-4">
                  {msg.text}
                </div>
              ) : (
                <div className={`max-w-[85%] md:max-w-[60%] flex flex-col ${msg.senderName === username ? "items-end" : "items-start"}`}>
                  {msg.senderName !== username && (
                    <span className={`text-[11px] font-bold mb-1 ml-2 ${msg.senderName === "AURA-OS" ? "text-cyan-400" : "text-emerald-500"}`}>
                      {msg.senderName} {msg.senderName === "AURA-OS" && "✓ AI"}
                    </span>
                  )}
                  <div className={`p-3 md:p-4 relative text-[15px] shadow-lg ${msg.senderName === username
                    ? "bg-emerald-600 text-white rounded-2xl rounded-tr-sm"
                    : msg.senderName === "AURA-OS"
                      ? "bg-[#0f1b2e] border border-cyan-500/30 text-cyan-50 rounded-2xl rounded-tl-sm"
                      : "bg-[#162032] border border-white/5 text-gray-200 rounded-2xl rounded-tl-sm"
                    }`}>
                    <div className="leading-relaxed break-words">
                      {msg.text}
                      {msg.isStreaming && <span className={`inline-block w-2 h-4 ml-1 animate-pulse ${msg.senderName === "AURA-OS" ? "bg-cyan-400" : "bg-emerald-400"}`}></span>}
                    </div>
                    <div className={`text-[9px] font-mono mt-2 opacity-50 flex items-center justify-end gap-1 ${msg.senderName === username ? "text-emerald-100" : "text-gray-400"}`}>
                      {msg.timestamp.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
                    </div>
                  </div>
                </div>
              )}
            </motion.div>
          ))}
          <div ref={messagesEndRef} />
        </AnimatePresence>
      </div>

      {/* Input */}
      <div className="p-4 bg-[#090d16] border-t border-white/5">
        <div className="relative flex items-center w-full bg-[#050810] border border-white/10 rounded-full p-1 shadow-inner focus-within:border-emerald-500/50 transition-all">
          <button className="w-12 h-12 rounded-full hover:bg-white/5 flex items-center justify-center text-gray-400"><Terminal className="w-5 h-5" /></button>
          <input
            type="text"
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => e.key === "Enter" && handleSend()}
            placeholder="Type an encrypted message..."
            className="flex-1 bg-transparent border-none outline-none px-2 text-white placeholder-gray-600 text-sm w-full"
          />
          <button onClick={handleSend} disabled={!input.trim()} className="w-12 h-12 bg-emerald-500 rounded-full text-white flex items-center justify-center hover:bg-emerald-400 disabled:opacity-50 disabled:hover:bg-emerald-500 transition-all shadow-lg shadow-emerald-500/20">
            <Send className="w-5 h-5 ml-1" />
          </button>
        </div>
      </div>
    </div>
  );
}

function VideoCallInterface({ socket, activeCall, myUsername, onEnd }: any) {
  const localVideoRef = useRef<HTMLVideoElement>(null);
  const remoteVideoRef = useRef<HTMLVideoElement>(null);
  const [callStatus, setCallStatus] = useState("Establishing secure line...");
  const peerRef = useRef<any>(null);
  const streamRef = useRef<MediaStream | null>(null);

  useEffect(() => {
    navigator.mediaDevices.getUserMedia({ video: true, audio: true }).then(stream => {
      streamRef.current = stream;
      if (localVideoRef.current) localVideoRef.current.srcObject = stream;

      const peer = new Peer({
        initiator: activeCall.isCaller,
        trickle: false,
        stream: stream
      });

      peer.on('signal', (data: any) => {
        if (activeCall.isCaller) {
          socket.emit('call_user', { userToCall: activeCall.userId, signalData: data, from: myUsername, callerName: myUsername });
        } else {
          socket.emit('answer_call', { signal: data, to: activeCall.userId });
        }
      });

      peer.on('stream', (remoteStream: MediaStream) => {
        setCallStatus("Connected. Encrypted E2EE Call.");
        if (remoteVideoRef.current) remoteVideoRef.current.srcObject = remoteStream;
      });

      peer.on('error', (err: any) => {
        console.error(err);
        setCallStatus("Connection failed.");
      });

      socket.on('call_accepted', (signal: any) => {
        setCallStatus("Connecting...");
        peer.signal(signal);
      });

      if (!activeCall.isCaller && activeCall.initialSignal) {
        peer.signal(activeCall.initialSignal);
      }

      peerRef.current = peer;
    }).catch(err => {
      setCallStatus("Camera/Microphone access denied.");
    });

    return () => {
      socket.off('call_accepted');
      if (streamRef.current) {
        streamRef.current.getTracks().forEach(track => track.stop());
      }
      if (peerRef.current) {
        peerRef.current.destroy();
      }
    };
  }, []);

  const endCall = () => {
    socket.emit("end_call", { to: activeCall.userId });
    onEnd();
  };

  return (
    <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} exit={{ opacity: 0 }} className="absolute inset-0 bg-[#050810]/95 backdrop-blur-lg z-40 flex flex-col items-center justify-center">
      <h2 className="text-2xl font-bold text-white mb-2">{activeCall.username}</h2>
      <p className="text-emerald-500 font-mono text-xs mb-8 flex items-center gap-2"><Lock className="w-3 h-3" /> {callStatus}</p>

      <div className="relative w-full max-w-4xl aspect-video bg-black rounded-3xl overflow-hidden border border-white/10 shadow-2xl">
        <video ref={remoteVideoRef} autoPlay playsInline className="w-full h-full object-cover"></video>

        <div className="absolute bottom-6 right-6 w-48 aspect-video bg-black rounded-xl overflow-hidden border-2 border-emerald-500 shadow-lg">
          <video ref={localVideoRef} autoPlay playsInline muted className="w-full h-full object-cover transform -scale-x-100"></video>
        </div>
      </div>

      <div className="mt-8 flex gap-6">
        <button onClick={endCall} className="w-16 h-16 rounded-full bg-red-500 hover:bg-red-600 text-white flex items-center justify-center shadow-[0_0_30px_rgba(239,68,68,0.4)] transition-all">
          <Phone className="w-8 h-8 rotate-[135deg]" />
        </button>
      </div>
    </motion.div>
  );
}
